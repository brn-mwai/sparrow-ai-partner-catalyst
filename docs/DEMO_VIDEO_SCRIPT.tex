\section{Production Notes}

\subsection{Screen Recording Setup}
\begin{itemize}[leftmargin=1.5em]
    \item Use 1920x1080 resolution
    \item Hide bookmarks bar and other browser UI
    \item Use incognito mode for clean interface
    \item Ensure good internet connection for live call demo
\end{itemize}

\subsection{Audio Setup}
\begin{itemize}[leftmargin=1.5em]
    \item Record voiceover separately with quality microphone
    \item Capture system audio for ElevenLabs voice during call
    \item Mix voiceover at -12dB, call audio at -6dB
    \item Add subtle background music (optional, very low)
\end{itemize}

\subsection{Editing Tips}
\begin{itemize}[leftmargin=1.5em]
    \item Cut out loading times, speed up transitions
    \item Add subtle zoom effects on key UI elements
    \item Use lower-thirds for key statistics
    \item Ensure text is readable at 1080p
\end{itemize}

\subsection{Call Demo Preparation}
\begin{itemize}[leftmargin=1.5em]
    \item Practice the call script 3-5 times before recording
    \item Generate persona ahead of time, note the name
    \item Have backup recording in case of technical issues
    \item Keep call under 45 seconds for demo
\end{itemize}
